```python
# Problem: Find Maximum and Minimum in an Array
# Time: O(n), Space: O(1)

def find_min_max(arr):
    min_val = max_val = arr[0]
    for num in arr:
        if num < min_val:
            min_val = num
        elif num > max_val:
            max_val = num
    return min_val, max_val

# Example
print(find_min_max([3, 1, 7, 4, 2]))  # Output: (1, 7)
```
